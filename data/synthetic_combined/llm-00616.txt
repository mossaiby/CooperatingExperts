[english] I need a stable way to execute SQL queries against a database and retrieve all resulting records.
[english] This function provides a robust wrapper for database operations. It uses context managers like closing() on the connection and cursor to ensure everything is properly cleaned up after execution.

[python] def get_records(self, sql, parameters=None):
        """
        Executes the sql and returns a set of records.

        :param sql: the sql statement to be executed (str) or a list of
            sql statements to execute
        :type sql: str or list
        :param parameters: The parameters to render the SQL query with.
        :type parameters: mapping or iterable
        """
        with closing(self.get_conn()) as conn:
            with closing(conn.cursor()) as cur:
                if parameters is not None:
                    cur.execute(sql, parameters)
                else:
                    cur.execute(sql)
                return cur.fetchall()
