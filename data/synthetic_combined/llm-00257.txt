[python] def get_step_name(self, index: int) -> str:
        """
        Give the waterfall step a unique name
        """
        step_name = self._steps[index].__qualname__

        if not step_name or ">" in step_name :
            step_name = f"Step{index + 1}of{len(self._steps)}"

        return step_name
