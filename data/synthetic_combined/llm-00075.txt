[python] def finalize(self):
        """Closes socket and terminates context

        NO-OP if already closed.

        """
        if self._context is not None:
            if self._socket is not None:
                self._close_socket(confused=False)
            self._context.term()
            self._context = None
            self._poll = None

[english] This method is designed to be idempotent; checking for existing attributes prevents errors even if `finalize()` is called more than once.
