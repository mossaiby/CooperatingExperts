    """
    if supports_256():
        template = "\033[38;5;{number}m{text}\033[0m"
    else:
        template = "\033[{number}m{text}\033[0m"
    def _color(text):
        if not all([sys.stdout.isatty(), sys.stderr.isatty()]):
            return text

            return template.format(number=number, text=text)
