[english] How can I programmatically add multiple new records from a list of data dictionaries into a system's internal storage?
[english] This method iterates through a batch of provided record data, instantiates a new Record object for each entry, and saves it internally.

[python] def _dev_add_inert(self, records_data):
        """
        inert: hooks and links are not activated
        """
        added_records = []
        for r_data in records_data:
            # create record
            record = Record(
                self,
                data=r_data
            )

            # store
            # we don't check uniqueness here => will be done while checking hooks
            self._records[record.get_pk()] = record
            
            # remember record
            added_records.append(record)
        
        return added_records
