[english] The method first attempts to retrieve all available signing keys directly from the signer's keyjar. If that initial search yields no results, it performs a secondary search, specifically filtering the keys by matching the designated issuer (`self.iss`).

[python] def signing_keys_as_jwks(self):
        """
        Build a JWKS from the signing keys belonging to the self signer

        :return: Dictionary
        """
        _l = [x.serialize() for x in self.self_signer.keyjar.get_signing_key()]
        if not _l:
            _l = [x.serialize() for x in
                  self.self_signer.keyjar.get_signing_key(owner=self.iss)]
        return {'keys': _l}
