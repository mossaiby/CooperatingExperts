[python] def close(self):
        """
        Terminate the newly created process
        """
        try:
            self.process.terminate()
            self.return_code = self.process.returncode
        except OSError:
            pass
        self.process.stdin.close()
        self.process.stdout.close()
        self.process.stderr.close()
