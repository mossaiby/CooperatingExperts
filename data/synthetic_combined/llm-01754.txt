[english] This function acts as an intermediary wrapper that utilizes external library logic to preprocess text. It first imports the necessary formatting tools from `trepan.lib` to apply styling based on current configuration settings.
<switch:python>
[python] def rst_msg(self, text, opts={}):
        """Convert ReStructuredText and run through msg()"""
        from trepan.lib.format import rst_text
        text = rst_text(text,
                        'plain' == self.debugger.settings['highlight'],
                        self.debugger.settings['width'])
        return self.msg(text)
