[english] This method acts as an interface to increment a metric. It first checks if the metric should be sent before packaging and sending the final counter request.

[python] def increment(self, name, count=1, rate=1):
        # type: (str, int, float) -> None
        """Increment a Counter metric"""

        if self._should_send_metric(name, rate):
            self._request(
                Counter(
                    self._create_metric_name_for_request(name),
                    int(count),
                    rate
                ).to_request()
            )
