[english] To check for inheritance, the function iterates through all known ancestors of the `child` class. It then uses a helper method (`is_subtype`) provided by astroid to see if any ancestor is considered a subtype of the given `parent` node.

[python] def is_subclass_of(child: astroid.ClassDef, parent: astroid.ClassDef) -> bool:
    """
    Check if first node is a subclass of second node.
    :param child: Node to check for subclass.
    :param parent: Node to check for superclass.
    :returns: True if child is derived from parent. False otherwise.
    """
    if not all(isinstance(node, astroid.ClassDef) for node in (child, parent)):
        return False

    for ancestor in child.ancestors():
        try:
            if astroid.helpers.is_subtype(ancestor, parent):
                return True

    return False
