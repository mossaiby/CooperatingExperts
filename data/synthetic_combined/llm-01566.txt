    """
    Parses response body from JSON, XML or query string.

    :param body:
        string

    :returns:

        :class:`xml.etree.ElementTree.Element` if XML.

    """

        # Try JSON first.
        return json.loads(body)
    except (OverflowError, TypeError, ValueError):
        pass

        # Then XML.
