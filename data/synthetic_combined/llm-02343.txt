[english] This function is designed for introspection, gathering every known location within the IPython environment that references namespaces. It compiles these locations into a single list of dictionary references.

[python] def all_ns_refs(self):
        """Get a list of references to all the namespace dictionaries in which
        IPython might store a user-created object.
        
        Note that this does not include the displayhook, which also caches
        objects from the output."""
        return [self.user_ns, self.user_global_ns,
                self._user_main_module.__dict__] + self._main_ns_cache.values()
