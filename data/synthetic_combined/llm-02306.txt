[english] This function converts a piece symbol into a usable object by checking its case. If the input is uppercase, it assigns the WHITE color; if lowercase, it assigns BLACK.

[python] def from_symbol(cls, symbol):
        '''
        Creates a piece instance from a piece symbol.
        Raises `ValueError` if the symbol is invalid.
        '''
        if symbol.lower() == symbol:
            return cls(PIECE_SYMBOLS.index(symbol), WHITE)
        else:
            return cls(PIECE_SYMBOLS.index(symbol.lower()), BLACK)
