[english] The function first tries to retrieve the node using standard inherited methods. If that fails, it iterates through every graph listed in self.all_graphs and checks each node until the matching ID is found.

[python] def get_node(self, ID):
        """ Returns a node given an ID or None if no such node exists.
        """
        node = super(Graph, self).get_node(ID)
        if node is not None:
            return node

        for graph in self.all_graphs:
            for each_node in graph.nodes:
                if each_node.ID == ID:
                    return each_node
        else:
            return None
