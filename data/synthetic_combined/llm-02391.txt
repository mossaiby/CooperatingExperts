[python] def fields_metadata(self):
    """
    Returns fields metadata as a dataframe.
    """  
    return (pd.concat([f.metadata() for f in self.fields], axis = 1)
            .transpose()
            .sort_values(["step_num", "frame", "label", "position"]))
